Do You Know Your ABCs?

Algorithm

There are only 7! = 5040 ways to arrange the numbers, so just try all combinations.

I know there is a much easier way to solve this (the smallest is A, the second smallest is B and C is the largest minus A and B), but this was the solution I came up with during the contest.

Python

def perm(l):
    res = []
    if len(l) == 0: return [[]]
    for i in range(len(l)):
        c = list(l)
        c.pop(i)
        res += [[l[i]]+j for j in perm(c)]
    return res

def check(nums):
    # A, B, C, A+B, B+C, C+A, and A+B+C
    a,b,c = nums[:3]
    if a<=b<=c:
        if a+b == nums[3]:
            if b+c == nums[4]:
                if c+a == nums[5]:
                    if a+b+c == nums[6]:
                        return True
    return False

perms = perm(list(map(int,input().split())))
for p in perms:
    if check(p):
        print(" ".join(map(str,p[:3])))
        break
