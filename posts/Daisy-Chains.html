Daisy Chains

Algorithm

Brute force - just look at all n^2 pictures taken and find if they have an average flower.

Python

def check(flowers):
    # Is there an average flower?
    avg = sum(flowers)/len(flowers)
    return (avg in flowers)
n = int(input())
flowers = list(map(int,input().split()))
num = 0
for i in range(n+1):
    for j in range(i):
        if check(flowers[j:i]):
            num += 1
print(num)
